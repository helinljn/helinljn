1. 安装CCSwitch
    https://github.com/farion1231/cc-switch/releases
    CC-Switch-v3.16.2-Windows.msi

2. 安装nodejs
    https://nodejs.cn/download
    node-v24.15.0-win-x64.zip

    配置环境变量：
        C:\ProgramData\node-v24.15.0-win-x64

    # 查看更新包
    npm outdated -g

    # 更新包
    npm i -g @anthropic-ai/claude-code@latest
    npm i -g @openai/codex@latest
    npm i -g @google/gemini-cli@latest
    npm i -g codewhale@latest

3. 配置codex
    C:\Users\helin\.codex\auth.json
        {
            "auth_mode": "apikey",
            "OPENAI_API_KEY": "xxx"
        }

    C:\Users\helin\.codex\config.toml
        model_provider = "custom"
        model = "gpt-5.5"
        model_reasoning_effort = "xhigh"
        model_context_window = 1000000
        model_auto_compact_token_limit = 900000
        cli_auth_credentials_store = "file"

        [model_providers.custom]
        name = "custom"
        wire_api = "responses"
        requires_openai_auth = true
        base_url = "https://www.right.codes/codex/v1"

        [windows]
        sandbox = "unelevated"

4. 配置gemini
    C:\Users\helin\.gemini\.env
        GOOGLE_GEMINI_BASE_URL=https://www.right.codes/gemini
        GEMINI_API_KEY=xxx
