1. 安装CCSwitch
    https://github.com/farion1231/cc-switch/releases
    CC-Switch-v3.14.1-Windows.msi

2. 安装nodejs
    https://nodejs.cn/download
    node-v24.15.0-win-x64.zip

    配置环境变量：
        C:\ProgramData\node-v24.15.0-win-x64

    npm i -g @anthropic-ai/claude-code@latest
    npm i -g @openai/codex@latest
    npm i -g deepseek@latest

3. 配置claude-code
    C:\Users\helin\.claude\settings.json
    {
        "env": {
            "ANTHROPIC_BASE_URL": "https://www.right.codes/claude-aws",
            "ANTHROPIC_AUTH_TOKEN": "xxx",
            "ANTHROPIC_MODEL": "claude-opus-4-6",
            "ANTHROPIC_DEFAULT_OPUS_MODEL": "claude-opus-4-6",
            "ANTHROPIC_DEFAULT_SONNET_MODEL": "claude-opus-4-6",
            "ANTHROPIC_DEFAULT_HAIKU_MODEL": "claude-opus-4-6",
            "CLAUDE_CODE_DISABLE_NONESSENTIAL_TRAFFIC": "1",
            "CLAUDE_AUTOCOMPACT_PCT_OVERRIDE": "90",
            "CLAUDE_CODE_EFFORT_LEVEL": "max",
            "API_TIMEOUT_MS": "600000"
        },
        "includeCoAuthoredBy": false,
        "theme": "dark"
    }

4. 配置codex
    C:\Users\helin\.codex\config.toml
        {
            "OPENAI_API_KEY": "xxx"
        }

        model_provider = "custom"
        model = "gpt-5.5"
        model_reasoning_effort = "high"
        disable_response_storage = true
        model_context_window = 1000000
        model_auto_compact_token_limit = 900000

        [model_providers]
        [model_providers.custom]
        name = "custom"
        wire_api = "responses"
        requires_openai_auth = true
        base_url = "https://www.right.codes/codex/v1"

        [windows]
        sandbox = "unelevated"
