# 显示现有源
conda config --show channels

# 清除现有源
conda config --remove-key channels

# 清除索引缓存
conda clean -i

# 恢复默认源
conda config --add channels defaults

# 更换conda源

# 环境变量
    Windows:
        用户变量Path中添加：
        C:\ProgramData\miniconda3\Scripts

    Linux:
        vi ~/.bashrc

        # conda
        export PATH=miniconda_dir/Scripts:$PATH

    然后打开终端(cmd & shell)执行：conda init
    后续就可以用：conda activate xxx

# Conda常用命令(如果更改base环境的话，则需要管理员权限)
    创建指定python版本的环境
    conda create -n py311 python=3.11.15

    删除指定python版本的环境
    conda env remove -n py311

    查看当前系统下的环境
    conda info -e

    激活某个环境
    conda activate py311

    退出某个环境
    conda deactivate

    查看指定环境下的包
    conda list -n py311

    在指定环境下安装包
    conda install -n py311 xxx

    在指定环境下卸载包
    conda uninstall -n py311 xxx

    在指定环境下升级包
    conda update -n py311 xxx

    更新指定环境下的所有包
    conda update -n py311 --all

    导出当前环境的所有依赖到文件(不导出构建号)
    conda env export --no-builds > environment.yml

    一键安装(重建环境)，根据yml文件创建新环境
    conda env create -f environment.yml

    根据yml文件更新环境
    conda env update -f environment.yml

如果遇到：raise ValueError("check_hostname requires server_hostname")报错，需要关闭翻墙代理
