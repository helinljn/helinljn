# 项目名字
SET(CURRENT_TARGET_NAME testpatch)
SET(CURRENT_SOURCE_DIR  ${CMAKE_CURRENT_SOURCE_DIR})

# 头文件目录
SET(CURRENT_PRIVATE_INCLUDE_DIRS
    ${CURRENT_SOURCE_DIR}
    ${CMAKE_PROJECT_ROOT_DIR}/src
    ${CMAKE_PROJECT_ROOT_DIR}/src/hcode
)

SET(CURRENT_PUBLIC_INCLUDE_DIRS
    # ...
)

# 宏定义、编译选项、链接库
IF(WIN32)
    # 宏定义
    SET(CURRENT_PRIVATE_COMPILE_DEFINITIONS
        -DTESTA_EXPORTS
    )

    SET(CURRENT_PUBLIC_COMPILE_DEFINITIONS
        # ...
    )

    # 编译选项
    SET(CURRENT_COMPILE_OPTIONS
        # ...
    )

    # 链接库
    SET(CURRENT_LINK_LIBS
        core
    )
ELSE()
    # 宏定义
    SET(CURRENT_PRIVATE_COMPILE_DEFINITIONS
        -DTESTA_EXPORTS
    )

    SET(CURRENT_PUBLIC_COMPILE_DEFINITIONS
        # ...
    )

    # 编译选项
    SET(CURRENT_COMPILE_OPTIONS
        # ...
    )

    # 链接库
    SET(CURRENT_LINK_LIBS
        core
    )
ENDIF()

# testpatch源文件
FILE(GLOB_RECURSE TESTPATCH_SRC_LIST
    ${CURRENT_SOURCE_DIR}/*.h
    ${CURRENT_SOURCE_DIR}/*.inl
    ${CURRENT_SOURCE_DIR}/*.hpp
    ${CURRENT_SOURCE_DIR}/*.c
    ${CURRENT_SOURCE_DIR}/*.cc
    ${CURRENT_SOURCE_DIR}/*.cpp
)

# 创建一个.def文件，只导出特定函数(只有Windows平台才有效)
IF(WIN32)
    SET(EXPORT_DEF_FILE "${CURRENT_SOURCE_DIR}/exports.def")
    FILE(WRITE  ${EXPORT_DEF_FILE} "EXPORTS\n")
    FILE(APPEND ${EXPORT_DEF_FILE} "    ?testb_func@@YA?AV?$basic_string@DU?$char_traits@D@std@@V?$allocator@D@2@@std@@H@Z\n")
    FILE(APPEND ${EXPORT_DEF_FILE} "    ?func1@testb_base@@UEBA?AV?$basic_string@DU?$char_traits@D@std@@V?$allocator@D@2@@std@@AEBV23@@Z\n")
    FILE(APPEND ${EXPORT_DEF_FILE} "    ?func1@testb@@UEBA?AV?$basic_string@DU?$char_traits@D@std@@V?$allocator@D@2@@std@@AEBV23@@Z\n")
    FILE(APPEND ${EXPORT_DEF_FILE} "    ?func2@testb@@QEBA?AV?$basic_string@DU?$char_traits@D@std@@V?$allocator@D@2@@std@@AEBV23@@Z\n")
    FILE(APPEND ${EXPORT_DEF_FILE} "    ?func3@testb@@SA?AV?$basic_string@DU?$char_traits@D@std@@V?$allocator@D@2@@std@@AEBV23@@Z\n")
ELSE()
    SET(EXPORT_DEF_FILE "")
ENDIF()

# 生成动态库
ADD_LIBRARY(${CURRENT_TARGET_NAME} SHARED ${TESTPATCH_SRC_LIST} ${EXPORT_DEF_FILE})
PROJECT_TARGET_APPLY_COMMON_OPTIONS(${CURRENT_TARGET_NAME})
TARGET_INCLUDE_DIRECTORIES(${CURRENT_TARGET_NAME} PRIVATE ${CURRENT_PRIVATE_INCLUDE_DIRS})
TARGET_INCLUDE_DIRECTORIES(${CURRENT_TARGET_NAME} PUBLIC  ${CURRENT_PUBLIC_INCLUDE_DIRS})
TARGET_COMPILE_DEFINITIONS(${CURRENT_TARGET_NAME} PRIVATE ${CURRENT_PRIVATE_COMPILE_DEFINITIONS})
TARGET_COMPILE_DEFINITIONS(${CURRENT_TARGET_NAME} PUBLIC  ${CURRENT_PUBLIC_COMPILE_DEFINITIONS})
TARGET_COMPILE_OPTIONS(${CURRENT_TARGET_NAME}     PRIVATE ${CURRENT_COMPILE_OPTIONS})
TARGET_LINK_LIBRARIES(${CURRENT_TARGET_NAME}      PUBLIC  ${CURRENT_LINK_LIBS})

# 拷贝构建产物至工作目录
PROJECT_TARGET_COPY_RUNTIME_TO_WORKDIR(${CURRENT_TARGET_NAME})

# 其它设置
IF(WIN32)
    # MSVC运行库设置
    SET_PROPERTY(TARGET ${CURRENT_TARGET_NAME} PROPERTY MSVC_RUNTIME_LIBRARY "MultiThreaded$<$<CONFIG:Debug>:Debug>DLL")

    # VS工程设置
    SET_PROPERTY(TARGET ${CURRENT_TARGET_NAME} PROPERTY FOLDER "projects/hcode")
    SET_PROPERTY(TARGET ${CURRENT_TARGET_NAME} PROPERTY VS_DEBUGGER_WORKING_DIRECTORY "${PROJECT_DEBUGGER_WORKING_DIRECTORY}")

    SOURCE_GROUP(TREE ${CURRENT_SOURCE_DIR} PREFIX "testpatch"
        FILES ${TESTPATCH_SRC_LIST})
ENDIF()
